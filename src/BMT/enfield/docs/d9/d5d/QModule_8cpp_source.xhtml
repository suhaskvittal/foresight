144 void efd::QModule::insertGate(NDGateSign::uRef gate) {
145  if (gate.get() == nullptr) {
146  ERR << "Trying to insert a 'nullptr' gate." << std::endl;
147  ExitWith(ExitCode::EXIT_unreachable);

149 
150  if (gate->getId() == nullptr) {

152  ExitWith(ExitCode::EXIT_unreachable);

155  std::string id = gate->getId()->getVal();
156 
157  if (mGatesMap.find(id) != mGatesMap.end()) {
158  std::cerr << "Replacing gate: '" << id << "'." << std::endl;
159 
160  for (auto it = mGates.begin(), end = mGates.end(); it != end; ++it) {
161  if ((*it)->getId()->getVal() == id) {
162  mGates.erase(it);
163  break;

165  }
166 
167  mGatesMap.erase(mGatesMap.find(id));
168  }
169

246 std::string efd::QModule::toString(bool pretty, bool printGates) const {
247  std::string str;
248 
249  if (mVersion.get() != nullptr)
250  str += mVersion->toString(pretty);
251 
252  for (auto& incl : mIncludes)
253  str += incl->toString(pretty);
254 
255  if (printGates) {
256  // Print all quantum gates.
257  for (auto gate : mGates)
258  str += gate->toString(pretty);
259  } else {
260  // Print those gates that are being used.
261  std::unordered_set<std::string> doPrint;
262 
263  for (auto& stmt : *mStatements) {
264  NDQOp::Ref qcall = nullptr;
265 
266  if (auto ifstmt = dynCast<NDIfStmt>(stmt.get())) {
267  // Get the QOp part (if it is a NDIfStmt node).
268  if (ifstmt->getQOp()->isGeneric()) {
269  qcall = ifstmt->getQOp();
270  }
271  } else {
272  qcall = dynCast<NDQOp>(stmt.get());
273  }
274 
275  if (qcall != nullptr) {
276  std::string gateId = qcall->getId()->getVal();
277 
278  // Print if it was not yet printed.
279  if (doPrint.find(gateId) == doPrint.end()) {
280  doPrint.insert(gateId);
281  }
282  }
283  }
284 
285  for (auto gate : mGates) {
286  if (doPrint.find(gate->getId()->getVal()) != doPrint.end() &&
287  !gate->isInInclude()) {
288  str += gate->toString(pretty);
289  }
290  }
291  }
292 
293  for (auto& reg : mRegs)
294  str += reg->toString(pretty);
295 
296  str += mStatements->toString(pretty);
297  return str;
298 }
