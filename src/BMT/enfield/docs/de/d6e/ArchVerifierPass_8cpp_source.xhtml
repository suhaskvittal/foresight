39 bool ArchVerifierVisitor::visitNDQOp(NDQOp::Ref qop) {
40  bool success;
41 
42  std::vector<uint32_t> qUIds;
43  for (auto& qarg : *qop->getQArgs()) {
44  auto qargStr = qarg->toString(false);
45 
46  if (mArch->hasSId(qargStr)) {
47  qUIds.push_back(mArch->getUId(qargStr));
48  } else {
49  // If there is some quantum operation that uses an inexistent qubit, we already
50  // may return false!
51  return false;
52  }
53  }
54 
55  if (IsCNOTGateCall(qop)) {
56  if (qUIds.size() != 2) {
57  ERR << "CNOT call must have 2 quantum qubits. Actual: `"
58  << qop->toString(false) << "`." << std::endl;
59  ExitWith(ExitCode::EXIT_unreachable);
60  }
61 
62  success = checkCNOT({ qUIds[0], qUIds[1] });
63 
64  } else if (IsIntrinsicGateCall(qop)) {
65  auto gen = dynCast<NDQOpGen>(qop);
66 
67  if (gen == nullptr) {
68  ERR << "Intrinsic call not generic operation!? Actual: `"
69  << qop->toString(false) << "`." << std::endl;
70  ExitWith(ExitCode::EXIT_unreachable);
71  }
72 
73  if (!gen->isIntrinsic()) {
74  ERR << "Intrinsic call not marked as intrinsic!? Actual: `"
75  << qop->toString(false) << "`." << std::endl;
76  ExitWith(ExitCode::EXIT_unreachable);
77  }
78 
79  switch (gen->getIntrinsicKind()) {
80  case NDQOpGen::K_INTRINSIC_SWAP:
81  if (qUIds.size() != 2) {
82  ERR << "SWAP call must have 2 quantum qubits. Actual: `"
83  << qUIds.size() << "`." << std::endl;
84  ExitWith(ExitCode::EXIT_unreachable);
85  }
86 
87  success = checkCNOT({ qUIds[0], qUIds[1] }) || checkCNOT({ qUIds[1], qUIds[0] });
88  break;
89 
90  case NDQOpGen::K_INTRINSIC_LCX:
91  if (qUIds.size() != 3) {
92  ERR << "LCX call must have 3 quantum qubits. Actual: `"
93  << qUIds.size() << "`." << std::endl;
94  ExitWith(ExitCode::EXIT_unreachable);
95  }
96 
97  success = (checkCNOT({ qUIds[0], qUIds[1] }) || checkCNOT({ qUIds[1], qUIds[0] })) &&
98  (checkCNOT({ qUIds[1], qUIds[2] }) || checkCNOT({ qUIds[2], qUIds[1] }));
99  break;
100 
101  case NDQOpGen::K_INTRINSIC_REV_CX:
102  if (qUIds.size() != 2) {
103  ERR << "Reversal call must have 2 quantum qubits. Actual: `"
104  << qUIds.size() << "`." << std::endl;
105  ExitWith(ExitCode::EXIT_unreachable);
106  }
107 
108  success = checkCNOT({ qUIds[1], qUIds[0] });
109  break;
110  }
111  } else {
112  success = true;
113  }
114 
115  return success;
116 }
