171 void CircuitGraph::init(uint32_t qubits, uint32_t cbits) {
172  mInit = true;
173  mQubits = qubits;
174  mCbits = cbits;
175 
176  mGraphHead.assign(mQubits + mCbits,
177  CircuitNode::sRef(new CircuitNode(CircuitNode::Type::INPUT)));
178  mGraphTail.assign(mQubits + mCbits,
179  CircuitNode::sRef(new CircuitNode(CircuitNode::Type::OUTPUT)));
180 
181  // Initializing the head to point to the tail, and the tail to the head.
182  for (uint32_t i = 0, e = mQubits + mCbits; i < e; ++i) {
183  mGraphHead[i]->mStepMap[i] = std::make_pair(nullptr, mGraphTail[i]);
184  mGraphTail[i]->mStepMap[i] = std::make_pair(mGraphHead[i], nullptr);
185  }
186 }
